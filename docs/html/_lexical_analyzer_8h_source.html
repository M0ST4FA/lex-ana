104 template<typename TokenT, typename TableT, typename InputT>
105 void LexicalAnalyzer<TokenT, TableT, InputT>::_remove_whitespace_prefix(unsigned flags)
106 {
107
108 // this function will be entered in case white space characters are to be removed.
109 // remove all whitespaces...
110 while (true) {
111 const char currChar = *this->m_SourceCode.data();
112 const bool isWhiteSpace = std::isspace(currChar) && currChar != '\0'; // null marks the EOF.
113 // test whether `currChar` is a new line and new lines are allowed to be counted
114 const bool matchNewLine = currChar == '\n' && (flags & (unsigned)LA_FLAG::LAF_ALLOW_NEW_LINE);
115
116 // if we have not catched a white space...
117 if (!isWhiteSpace)
118 return; // early return
119
120 // erase whitespace from source code stream
121 this->m_SourceCode.remove_prefix(1);
122
123 // if the current character is a new line char and they are allowed, do not remove the current char
124 if (matchNewLine)
125 this->m_Line++, this->m_Col = 0;
126 else
127 this->m_Col++; // if this whitespace character is not a new line character
128 }
129
130 }

142 {
143
144 // if checks fail
145 if (not this->_check_presearch_conditions(flags))
146 return Result{};
147
148 // check whether there is a matched lexeme
149 const FSMResult fsmRes = this->m_Automatan.simulate(this->m_SourceCode, FSM_MODE::MM_LONGEST_PREFIX);
150
151 // if there is not, early return
152 if (not fsmRes.accepted)
153 return LexicalAnalyzerResult<TokenT, InputT>{};
154
155 // there is a matching lexeme
156
157 // 1. extract it
158 const Result res{ fsmRes, this->getLine(), this->getCol(), this->m_TokenFactory};
159
160 // 2. adjust the column (character) number
161 const size_t lexemeSize = fsmRes.size();
162 this->m_Col += lexemeSize;
163
164 // 3. erase the lexeme from source code stream
165 m_SourceCode.remove_prefix(lexemeSize);
166
167 return res;
168 }
