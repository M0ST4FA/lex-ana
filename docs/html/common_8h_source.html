189 bool operator==(const IterableT& lhs, const IterableT& rhs) {
190 // if they do not have the same size, they are not coequal
191 if (lhs.size() != rhs.size())
192 return false;
193
194 // if they have the same number of elements
195
196 // if, for every element `a` in `lhs`, `a` is also an element
197 // of `rhs`, then they are coequal
198
199 for (const auto& lhsElement : lhs) {
200 // check whether `rhs` contains that element
201 if (rhs.contains(lhsElement))
202 continue;
203
204 // if `rhs` does not contain that element
205 return false;
206 }
207
208 return true;

218 bool operator==(const IterableT& lhs, const IterableT& rhs) {
219 const size_t lhsSize = lhs.size();
220 const size_t rhsSize = rhs.size();
221
222 // if they have different sizes, they are not equal
223 if (lhsSize != rhsSize)
224 return false;
225
226 // if there exists at least a single element in `lhs`
227 // that does not exist in `rhs`, in order, they are not equal
228 for (size_t i = 0; i < lhsSize; i++) {
229 const auto& lhsElement = lhs.at(i);
230 const auto& rhsElement = rhs.at(i);
231
232 // if the current elements in each prod body are not equal,
233 // the two productions are not equal
234 if (lhsElement != rhsElement)
235 return false;
236 }
237
238 // if we reach here, it means the two iterables are not `not-equal`
239 // hence they must be equal
