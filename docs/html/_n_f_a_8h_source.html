92 FSMResult NonDeterFiniteAutomatan<TransFuncT, InputT>::_simulate_longest_prefix(const InputT& input) const
93 {
94 constexpr FSMStateType startState = FiniteStateMachine<TransFuncT, InputT>::START_STATE;
95 std::vector<FSMStateSetType> matchedStates = { {startState} };
96 size_t charIndex = 0;
97
107 if (this->getMachineType() == FSMType::MT_NON_EPSILON_NFA)
108 for (; charIndex < input.size(); charIndex++) {
109 const auto c = input.at(charIndex);
110 // get next set of states and update our path through the machine
111 matchedStates.push_back(this->m_TransitionFunc(matchedStates.back(), c));
112 }
113 else
114 for (; charIndex < input.size(); charIndex++) {
115 const auto c = input.at(charIndex);
116 // get next set of states and update our path through the machine
117 matchedStates.push_back(this->_epsilon_closure(this->m_TransitionFunc(matchedStates.back(), c)));
118 }
119
120 // make sure `charIndex` < input.size()
121 if (charIndex == input.size())
122 charIndex = input.size() - 1;
123
124 // set the index of the current character; figure out whether there is an accepted longest prefix; set the end index
125 bool accepted = _check_accepted_longest_prefix(matchedStates, charIndex);
126 size_t end = accepted ? charIndex + 1 : 0;
127
128 // get the final states we've reached
129 const FSMStateSetType& finalStates = this->_get_final_states_from_state_set(matchedStates.back());
130

197 {

199 assert(charIndex == startIndex);
200
205 auto matchInput = [input, &charIndex, &matchedStates, this](bool calcClosure = false) {
206 for (; charIndex < input.size(); charIndex++) {
207
208 auto c = input[charIndex];

210 FSMStateSetType currStateSet{};

306 template<typename TransFuncT, typename InputT>
307 FSMStateSetType NonDeterFiniteAutomatan<TransFuncT, InputT>::_epsilon_closure(const FSMStateSetType& set) const
308 {
309 FSMStateSetType res{set};
310
311 std::stack<FSMStateType> stack{};
312 // initialize the stack
313 for (auto s : set)
314 stack.push(s);
315
316 while (stack.size()) {
317 // get the last state and pop it; we will get its closure now, so we will not need it in the future
318 FSMStateType s = stack.top();
319 stack.pop();
320
321 // check if the state has at least one transition on epsilon
322 FSMStateSetType epsilonTransitions = this->m_TransitionFunc(s, '\0');
323
324 // if the set has at least a single epsilon-transition
325 if (!epsilonTransitions.empty()) {
333 for (const FSMStateType state : epsilonTransitions)
334 if (!res.contains(state))
335 stack.push(state);
336
337 res.insert(epsilonTransitions.begin(), epsilonTransitions.end());
338 };
339
340 };
341
342 return res;
343 }
